## OAuth with PKCE

```ts
import { createPkceFlow } from "@heb-sdk-unofficial/auth"

const flow = await createPkceFlow({
  redirectUri: "https://your-app.dev/callback",
})

console.log(flow.authorizeUrl)
```

## Token storage

```ts
await tokenStore.save({
  accessToken: "example-access-token",
  refreshToken: "example-refresh-token",
  expiresAt: Date.now() + 3600 * 1000,
})
```

## Cookie sessions

```ts
const client = createClient({
  session: { cookie: "example-cookie" },
  store: { id: "123" },
})
```
